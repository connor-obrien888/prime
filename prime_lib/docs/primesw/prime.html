150    def predict(self, input = None, start = None, stop = None, pos = [13.25, 0, 0]):
151        '''
152        Generate prime predictions from input dataframes or time ranges.
153        
154        Parameters:
155            input (dataframe, ndarray): Input data to be scaled and predicted
156            start (string, optional): Start time of prediction (will use real data). Format 'YYYY-MM-DD HH:MM:SS'.
157            stop (string, optional): Stop time of prediction (will use real data). Format 'YYYY-MM-DD HH:MM:SS'.
158            pos (list, optional): Position propagated to if 'start' and 'stop' are specified.
159        Returns:
160            output (dataframe): Scaled output data
161        '''
162        if input is None:
163            if (start is not None)&(stop is not None):
164                input = self.build_real_input(start = start, stop = stop, pos = pos)
165            else:
166                raise RuntimeWarning('Must specify either input or (start and stop).')
167                return None
168        if isinstance(input, pd.DataFrame): # If input is a dataframe
169            input_arr = input[self.in_keys].to_numpy() # Convert input dataframe to array
170        if isinstance(input, np.ndarray): # If input is an array
171            input_arr = input # Set input array to input
172        output_arr = self.predict_raw(input_arr) # Predict with the keras model
173        output = pd.DataFrame(output_arr, columns = self.out_keys) # Convert output array to dataframe
174        output_epoch = input['Epoch'].to_numpy()[(self.window-1):] # Stage an epoch column to be added to the output dataframe
175        output_epoch += pd.Timedelta(seconds = 100*self.stride) # Add lead time to the epoch column
176        output['Epoch'] = output_epoch # Add the epoch column to the output dataframe

178    def predict_raw(self, input):
179        '''
180        Generates PRIME predictions from input dataframe. Assumes that input has keys specified by `primesw.prime.in_keys`. It is generally recommended to use `primesw.prime.predict` instead.
181        '''
182        input_scaled = self.in_scaler.transform(input) # Rescale the input data
183        input_arr = np.zeros((len(input_scaled)-(self.window-1), self.window, len(self.in_keys))) # Reshape input data to be 3D
184        for i in np.arange(len(input_scaled)-(self.window-1)):
185            input_arr[i,:,:] = input_scaled[i:(i+self.window)] # Move the 55 unit window through the input data
186        output_unscaled = self.model.predict(input_arr) # Use stored keras model to make prediction
187        output = np.zeros((len(output_unscaled),len(self.out_keys))) #Stage output data to be 2x target dimensions (to account for uncertainties)
188        output[:, ::2] = self.tar_scaler.inverse_transform(output_unscaled[:, ::2]) #Mean values
189        output[:, 1::2] = np.abs(self.tar_scaler.inverse_transform(output_unscaled[:, ::2] + output_unscaled[:, 1::2]) - self.tar_scaler.inverse_transform(output_unscaled[:, ::2])) #Standard deviations
190        return output
191    def predict_grid(
192        self,
193        gridsize,
194        x_extent,
195        framenum,
196        bx,
197        by,
198        bz,
199        vx,
200        vy,
201        vz,
202        ni,
203        vt,
204        rx,
205        ry,
206        rz,
207        y_extent=None,
208        z_extent=None,
209        y = 0,
210        z = 0,
211        subtract_ecliptic=False,
212    ):
213        """
214        Generate predictions from prime model on a grid of points.
215
216        Parameters:
217            gridsize (float): Spacing of grid points
218            x_extent (list): Range of x values to calculate on
219            framenum (int): Number of frames to calculate
220            bx (float, array-like): IMF Bx value. If array like, must be of length framenum.
221            by (float, array-like): IMF By value. If array like, must be of length framenum.
222            bz (float, array-like): IMF Bz value. If array like, must be of length framenum.
223            vx (float, array-like): Solar wind Vx value. If array like, must be of length framenum.
224            vy (float, array-like): Solar wind Vy value. If array like, must be of length framenum.
225            vz (float, array-like): Solar wind Vz value. If array like, must be of length framenum.
226            ni (float, array-like): Solar wind ion density value. If array like, must be of length framenum.
227            vt (float, array-like): Solar wind ion thermal speed value. If array like, must be of length framenum.
228            rx (float, array-like): Wind spacecraft position x value. If array like, must be of length framenum.
229            ry (float, array-like): Wind spacecraft position y value. If array like, must be of length framenum.
230            rz (float, array-like): Wind spacecraft position z value. If array like, must be of length framenum.
231            y_extent (list): Range of y values to calculate on. If None, z_extent must be specified.
232            z_extent (list): Range of z values to calculate on. If None, y_extent must be specified.
233            y (float, array-like): Y position that is held constant if y_extent is not specified. Default 0.
234            z (float, array-like): Z position that is held constant if z_extent is not specified. Default 0.
235            subtract_ecliptic (bool): Whether or not to subtract the Earth's motion in the ecliptic from Vy
236        Returns:
237            output_grid (ndarray): Array of predicted values on the grid. Shape (framenum, x_extent/gridsize, y_extent/gridsize, 18)
238        """
239        x_arr = np.arange(x_extent[0], x_extent[1], gridsize)  # Create a grid to calculate the solar wind conditions on
240        y_arr = np.asarray([y]) # This array is overwritten if y_extent is specified
241        z_arr = np.asarray([z]) # This array is overwritten if z_extent is specified
242        if y_extent is None and z_extent is None:
243            raise ValueError("Must specify y_extent or z_extent")
244        if y_extent is not None:
245            y_arr = np.arange(y_extent[0], y_extent[1], gridsize)  # Y positions to calculate the solar wind conditions on
246        if z_extent is not None:
247            z_arr = np.arange(z_extent[0], z_extent[1], gridsize)  # Z positions to calculate the solar wind conditions on
248        x_grid, y_grid, z_grid = np.meshgrid(x_arr, y_arr, z_arr)  # Create a grid to calculate the solar wind conditions on
249        input_seed = np.zeros((len(x_grid.flatten()) * framenum, len(self.in_keys)))  # Initialize array to hold the input data before unfolding it
250        for idx, element in enumerate([bx, by, bz, vx, vy, vz, ni, vt, rx, ry, rz]):  # Loop through the input data and repeat it
251            try:
252                iter(element)  # Check if the element is iterable
253                input_seed[:, idx] = np.repeat(element, len(x_grid.flatten()))  # If it is, repeat it for each grid point
254            except TypeError:  # This error throws if iter(element) fails (i.e. element is not iterable)
255                input_seed[:, idx] = np.repeat(element, framenum * len(x_grid.flatten()))  # If it isn't, repeat it for each grid point *and frame*
256        loc_arr = np.zeros((len(x_grid.flatten()) * framenum, 3))  # Initialize array to hold the location data
257        loc_arr[:, 0] = np.tile(x_grid.flatten(), framenum)
258        loc_arr[:, 1] = np.tile(y_grid.flatten(), framenum)
259        loc_arr[:, 2] = np.tile(z_grid.flatten(), framenum)
260        input_seed_scaled = self.in_scaler.transform(input_seed)  # Scale the input data
261        input_seed_scaled[:, 11:14] = self.loc_scaler.transform(loc_arr)  # Scale the location data
262        input_seed_scaled = np.repeat(input_seed_scaled, self.window, axis=0)  # Repeat the input data 55 times to make static timeseries
263        input_arr = input_seed_scaled.reshape(len(x_grid.flatten()) * framenum, self.window, len(self.in_keys))  # Reshape the input data into the correct shape
264        output_arr = self.model.predict(input_arr)  # Predict the output data
265        output = np.zeros((len(output_arr), len(self.out_keys)))  # Stage output data to be 2x target dimensions
266        output[:, ::2] = self.tar_scaler.inverse_transform(output_arr[:, ::2])  # Mean values
267        output[:, 1::2] = np.abs(self.tar_scaler.inverse_transform(output_arr[:, ::2] + output_arr[:, 1::2]) - self.tar_scaler.inverse_transform(output_arr[:, ::2]))  # Standard deviations
268        output_grid = output.reshape(framenum, len(y_arr), len(x_arr), len(z_arr), len(self.out_keys))  # Reshape the output data into the correct shape
269        output_grid = np.swapaxes(output_grid, 1, 2)  # Move the y axis to the second axis (new order is frame, x, y, z, param)
270        if subtract_ecliptic:  # If subtract_ecliptic is true, subtract the Earth's motion in the ecliptic from Vy
271            output_grid[:, :, :, :, 8] -= 29.8
272        return output_grid
273    def build_model(self, units = [352, 192, 48, 48], activation = 'elu', dropout = 0.20, lr = 1e-4):
274        '''
275        Builds the underlying PRIME model with no weights or biases loaded.
276
277        Parameters:
278            units (list): Number of units in each layer of the model
279            activation (str): Activation function to use in hidden layers
280            dropout (float): Dropout rate to use in hidden layers
281            lr (float): Learning rate to use in optimizer
282        
283        Returns:
284            model (keras model): Keras model to be used for prediction (weights not initialized)
285        '''
286        model = ks.Sequential([ks.layers.GRU(units=units[0]),
287                               ks.layers.Dense(units=units[1], activation=activation),
288                               ks.layers.Dense(units=units[2], activation=activation),
289                               ks.layers.Dense(units=units[3], activation=activation),
290                               ks.layers.LayerNormalization(),
291                               ks.layers.Dropout(dropout),
292                               ks.layers.Dense(len(self.tar_keys),activation='linear')
293                               ])
294        model.compile(optimizer=tf.optimizers.Adamax(learning_rate=lr), loss=crps_loss)
295        model.build(input_shape = (1, self.window, len(self.in_keys)))
296        return model

298            self,

300            bx = 0,

315        '''
316        Builds a synthetic input array from user-specified quantities at L1.
317        For input arrays made from measured data at L1, see `primesw.prime.build_real_input`.
318        
319        Parameters:
320            epoch (datetime): Datetime of start of input Dataframe.
321            bx (float, array-like): IMF Bx value.
322            by (float, array-like): IMF By value.
323            bz (float, array-like): IMF Bz value.
324            vx (float, array-like): Solar wind Vx value.
325            vy (float, array-like): Solar wind Vy value.
326            vz (float, array-like): Solar wind Vz value.
327            ni (float, array-like): Solar wind ion density value.
328            vt (float, array-like): Solar wind ion thermal speed value.
329            rx (float, array-like): Wind spacecraft position x value.
330            ry (float, array-like): Wind spacecraft position y value.
331            rz (float, array-like): Wind spacecraft position z value.
332            tar_rx (float, array-like): Propagation target position x value.
333            tar_ry (float, array-like): Propagation target position y value.
334            tar_rz (float, array-like): Propagation target position z value.
335        
336        Returns:
337            input (Dataframe): Input dataframe suitable to predict from with self.predict(). 
338        '''
339        input = pd.DataFrame(columns = self.in_keys) #Initialize single-point input dataframe
340        input['B_xgsm'] = bx*np.ones(self.window) #SW BX in nT (GSM coordinates)
341        input['B_ygsm'] = by*np.ones(self.window) #SW BY in nT (GSM coordinates)
342        input['B_zgsm'] = bz*np.ones(self.window) #SW BZ in nT (GSM coordinates)
343        input['Vi_xgse'] = vx*np.ones(self.window) #SW X velocity in km/s (GSE coordinates)
344        input['Vi_ygse'] = vy*np.ones(self.window) #SW Y velocity in km/s (GSE coordinates)
345        input['Vi_zgse'] = vz*np.ones(self.window) #SW Z velocity in km/s (GSE coordinates)
346        input['Ni'] = ni*np.ones(self.window) #SW density in cm^-3
347        input['Vth'] = ni*np.ones(self.window) #SW thermal velocity in km/s
348        input['R_xgse'] = rx*np.ones(self.window) #Wind position in X GSE (RE)
349        input['R_ygse'] = ry*np.ones(self.window) #Wind position in Y GSE
350        input['R_zgse'] = rz*np.ones(self.window) #Wind position in Z GSE
351
352        #This is where the location you're propagating to is set.
353        #Each entry in the "timeseries" is set to the same value, no need to make a fake MMS orbit.
354        input['target_R_xgse'] = tar_rx*np.ones(self.window) #MMS position in X GSE (RE)
355        input['target_R_ygse'] = tar_ry*np.ones(self.window) #MMS position in Y GSE
356        input['target_R_zgse'] = tar_rz*np.ones(self.window) #MMS position in Z GSE
357
358        #PRIME-SH also currently expects an 'Epoch' for its inputs.
359        input['Epoch'] = pd.date_range(start=epoch, periods=self.window, freq='100s') - pd.Timedelta(seconds = 100*(self.window + self.stride))
360        return input

362    def build_real_input(self, start, stop, pos =[13.25, 0, 0], load_freq = '3M'):
363        '''
364        Load Wind spacecraft input data for PRIME in between specified date strings.
365
366        Parameters:
367            start (string): The start date of the data to load ('YYYY-MM-DD')
368            end (string): The end date of the data to load ('YYYY-MM-DD')
369            pos (list): Location of propagation in GSE coordinates (Earth Radii). Default [13.25, 0, 0].
370            load_freq (string): Max length of data loaded by CdasWs. If throwing RuntimeError, try modifying this parameter. Default '3M' (three months).
371        Return:
372            input (Dataframe): Input dataframe suitable to predict from with self.predict(). 
373        '''
374        try:
375            from cdasws import CdasWs
376        except:
377            raise RuntimeError('Unable to import CdasWs. Predicting using non-synthetic input requires CdasWs package (see https://cdaweb.gsfc.nasa.gov/WebServices/REST/py/cdasws/).')
378        cdas = CdasWs() #Initialize CDAS WS Session
379        mfi_df = pd.DataFrame([]) #Staging dataframe for Wind spacecraft Magnetic Field Investigation data
380        try:
381            data = cdas.get_data('WI_H0_MFI', ['BGSM', 'PGSE'], start, stop) #Load GSM B field and GSE SC position
382            mfi_df['Epoch'] = data[1]['Epoch'] #MFI timestamps
383            mfi_df['R_xgse'] = data[1]['PGSE'][:, 0] #Wind SC position
384            mfi_df['R_ygse'] = data[1]['PGSE'][:, 1]
385            mfi_df['R_zgse'] = data[1]['PGSE'][:, 2]
386            mfi_df['B_xgsm'] = data[1]['BGSM'][:, 0] #GSM B field
387            mfi_df['B_ygsm'] = data[1]['BGSM'][:, 1]
388            mfi_df['B_zgsm'] = data[1]['BGSM'][:, 2]
389        except TypeError: #Throws when date range is empty OR too big
390            raise RuntimeError('CDASWS failed to load MFI data. Date range ('+start+' to '+stop+') may be too large or data may be missing.')
391        mfi_df['Epoch'] = pd.to_datetime(mfi_df['Epoch'], utc=True) #Convert to UTC aware datetime
392        #Set B values to nan if they are equal to the fill value of -1e31

467def crps_loss(y_true, y_pred):
468    """
469    Tensorflow implementation of the Continuous Rank Probability Score loss function. Assumes seven output features. For a simpler functional version, see `primesw.crps_f`.
470    
471    Parameters
472    ----------
473    y_true : tf.Tensor
474        Ground truth values of predicted variable.
475    y_pred : tf.Tensor
476        mu and sigma^2 values of predicted distribution.
477        
478    Returns
479    -------
480    crps : tf.Tensor
481        Continuous rank probability score.
482    """
483    # Separate the parameters into means and squared standard deviations
484    mu0, sg0, mu1, sg1, mu2, sg2, mu3, sg3, mu4, sg4, mu5, sg5, mu6, sg6, y_true0, y_true1, y_true2, y_true3, y_true4, y_true5, y_true6 = unstack_helper(y_true, y_pred)
485    
486    # CRPS (assuming gaussian distribution)
487    crps0 = tf.math.reduce_mean(crps_f(ep_f(y_true0, mu0), sg0))
488    crps1 = tf.math.reduce_mean(crps_f(ep_f(y_true1, mu1), sg1))
489    crps2 = tf.math.reduce_mean(crps_f(ep_f(y_true2, mu2), sg2))
490    crps3 = tf.math.reduce_mean(crps_f(ep_f(y_true3, mu3), sg3))
491    crps4 = tf.math.reduce_mean(crps_f(ep_f(y_true4, mu4), sg4))
492    crps5 = tf.math.reduce_mean(crps_f(ep_f(y_true5, mu5), sg5))
493    crps6 = tf.math.reduce_mean(crps_f(ep_f(y_true6, mu6), sg6))
494    
495    # Average the continuous rank probability scores
496    crps = (crps0 + crps1 + crps2 + crps3 + crps4 + crps5 + crps6) / 7.0
497    
498    return crps

500def crps_f(ep, sg):
501    '''
502    Helper function that calculates continuous rank probability score.
503    '''
504    crps = sg * ((ep/sg) * tf.math.erf((ep/(np.sqrt(2)*sg))) + tf.math.sqrt(2/np.pi) * tf.math.exp(-ep**2 / (2*sg**2)) - 1/tf.math.sqrt(np.pi))
505    return crps

507def ep_f(y, mu):
508    '''
509    Helper function that calculates epsilon (error) for CRPS.
510    '''
511    ep = tf.math.abs(y - mu)
512    return ep

225            vy (float, array-like): Solar wind Vy value. If array like, must be of length framenum.
226            vz (float, array-like): Solar wind Vz value. If array like, must be of length framenum.

238            output_grid (ndarray): Array of predicted values on the grid. Shape (framenum, x_extent/gridsize, y_extent/gridsize, 18)
239        """
240        x_arr = np.arange(x_extent[0], x_extent[1], gridsize)  # Create a grid to calculate the solar wind conditions on
241        y_arr = np.asarray([y]) # This array is overwritten if y_extent is specified
242        z_arr = np.asarray([z]) # This array is overwritten if z_extent is specified
243        if y_extent is None and z_extent is None:
244            raise ValueError("Must specify y_extent or z_extent")
245        if y_extent is not None:
246            y_arr = np.arange(y_extent[0], y_extent[1], gridsize)  # Y positions to calculate the solar wind conditions on
247        if z_extent is not None:
248            z_arr = np.arange(z_extent[0], z_extent[1], gridsize)  # Z positions to calculate the solar wind conditions on
249        x_grid, y_grid, z_grid = np.meshgrid(x_arr, y_arr, z_arr)  # Create a grid to calculate the solar wind conditions on
250        input_seed = np.zeros((len(x_grid.flatten()) * framenum, len(self.in_keys)))  # Initialize array to hold the input data before unfolding it
251        for idx, element in enumerate([bx, by, bz, vx, vy, vz, ni, vt, rx, ry, rz]):  # Loop through the input data and repeat it
252            try:
253                iter(element)  # Check if the element is iterable
254                input_seed[:, idx] = np.repeat(element, len(x_grid.flatten()))  # If it is, repeat it for each grid point
255            except TypeError:  # This error throws if iter(element) fails (i.e. element is not iterable)
256                input_seed[:, idx] = np.repeat(element, framenum * len(x_grid.flatten()))  # If it isn't, repeat it for each grid point *and frame*
257        loc_arr = np.zeros((len(x_grid.flatten()) * framenum, 3))  # Initialize array to hold the location data
258        loc_arr[:, 0] = np.tile(x_grid.flatten(), framenum)
259        loc_arr[:, 1] = np.tile(y_grid.flatten(), framenum)
260        loc_arr[:, 2] = np.tile(z_grid.flatten(), framenum)
261        input_seed_scaled = self.in_scaler.transform(input_seed)  # Scale the input data
262        input_seed_scaled[:, 11:14] = self.loc_scaler.transform(loc_arr)  # Scale the location data
263        input_seed_scaled = np.repeat(input_seed_scaled, self.window, axis=0)  # Repeat the input data 55 times to make static timeseries
264        input_arr = input_seed_scaled.reshape(len(x_grid.flatten()) * framenum, self.window, len(self.in_keys))  # Reshape the input data into the correct shape
265        output_arr = self.model.predict(input_arr)  # Predict the output data
266        output = np.zeros((len(output_arr), len(self.out_keys)))  # Stage output data to be 2x target dimensions
267        output[:, ::2] = self.tar_scaler.inverse_transform(output_arr[:, ::2])  # Mean values
268        output[:, 1::2] = np.abs(self.tar_scaler.inverse_transform(output_arr[:, ::2] + output_arr[:, 1::2]) - self.tar_scaler.inverse_transform(output_arr[:, ::2]))  # Standard deviations
269        output_grid = output.reshape(framenum, len(y_arr), len(x_arr), len(z_arr), len(self.out_keys))  # Reshape the output data into the correct shape
270        output_grid = np.swapaxes(output_grid, 1, 2)  # Move the y axis to the second axis (new order is frame, x, y, z, param)
271        if subtract_ecliptic:  # If subtract_ecliptic is true, subtract the Earth's motion in the ecliptic from Vy
272            output_grid[:, :, :, :, 8] -= 29.8
273        return output_grid
